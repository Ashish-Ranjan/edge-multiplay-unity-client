317  void HandleWebSocketMessage(string message)
318  {
319  var msg = MessageWrapper.UnWrapMessage(message);
320  switch (msg.type)
321  {
322  case "register":
323  Register register = Messaging<Register>.Deserialize(message);
324  gameSession.sessionId = register.sessionId;
325  gameSession.playerId = register.playerId;
326  EdgeMultiplayCallbacks.registerEvent();
327  break;
328 
329  case "notification":
330  Notification notification = Messaging<Notification>.Deserialize(message);
331  switch (notification.notificationText)
332  {
333  case "left-room":
334  gameSession.roomId = "";
335  EdgeMultiplayCallbacks.leftRoom();
336  break;
337  case "join-room-faliure":
338  EdgeMultiplayCallbacks.joinRoomFaliure();
339  break;
340 
341  }
342  if (notification.notificationText == "left-room")
343  {
344  EdgeMultiplayCallbacks.leftRoom();
345  }
346  if (notification.notificationText == "left-room")
347  {
348  EdgeMultiplayCallbacks.leftRoom();
349  }
350  EdgeMultiplayCallbacks.notificationEvent(notification);
351  break;
352 
353  case "roomsList":
354  RoomsList roomsList = Messaging<RoomsList>.Deserialize(message);
355  EdgeMultiplayCallbacks.roomsList(roomsList.rooms);
356  break;
357 
358  case "roomCreated":
359  RoomCreated roomCreated = Messaging<RoomCreated>.Deserialize(message);
360  gameSession.roomId = roomCreated.room.roomId;
361  EdgeMultiplayCallbacks.roomCreated(roomCreated.room);
362  break;
363 
364  case "roomJoin":
365  RoomJoin roomJoin = Messaging<RoomJoin>.Deserialize(message);
366  gameSession.roomId = roomJoin.room.roomId;
367  EdgeMultiplayCallbacks.roomJoin(roomJoin.room);
368  break;
369 
370  case "playerJoinedRoom":
371  PlayerJoinedRoom playerJoinedRoom = Messaging<PlayerJoinedRoom>.Deserialize(message);
372  EdgeMultiplayCallbacks.playerRoomJoined(playerJoinedRoom.room);
373  break;
374 
375  case "gameStart":
376  GameStart gameStart = Messaging<GameStart>.Deserialize(message);
377  gameSession.currentPlayers = gameStart.room.roomMembers.ToArray();
378  foreach (Player player in gameStart.room.roomMembers)
379  {
380  if (player.playerId == gameSession.playerId)
381  {
382  gameSession.playerIndex = player.playerIndex;
383  }
384  }
385  CreatePlayers(gameSession.currentPlayers);
386  gameStarted = true;
387  EdgeMultiplayCallbacks.gameStart();
388  print("gameSession.roomId : " + gameSession.roomId);
389  //room id received from EdgeMultiplayServer is "room-workerId-RoomIndex-maxPlayersPerRoom"
390  int workerIndex = 0;
391  int roomIndex = 0;
392  string[] roomDetails = gameSession.roomId.Split('-');
393  int.TryParse(roomDetails[1], out workerIndex);
394  int.TryParse(roomDetails[2], out roomIndex);
395  int udpReceivePort = 5000 + (workerIndex * 100) + (roomIndex * 10) + gameSession.playerIndex;
396  //udpClient = new MobiledgeXUDPClient("localhost", 5000, udpReceivePort);
397  udpClient = new MobiledgeXUDPClient(integration.GetHost(), integration.GetAppPort(LProto.L_PROTO_UDP).public_port, udpReceivePort);
398  SendUDPMessage(new GamePlayEvent
399  {
400  stringData = new string[] { "Start" }
401  });
402  break;
403 
404  case "GamePlayEvent":
405  GamePlayEvent mobiledgexEvent = Messaging<GamePlayEvent>.Deserialize(message);
406  ReflectEvent(mobiledgexEvent);
407  break;
408 
409  case "memberLeft":
410  PlayerLeft playerLeft = Messaging<PlayerLeft>.Deserialize(message);
411  EdgeMultiplayCallbacks.playerLeft(playerLeft);
412  break;
413 
414  case "NewRoomCreatedOnServer":
415  NewRoomCreatedOnServer newRoomCreatedOnServer = Messaging<NewRoomCreatedOnServer>.Deserialize(message);
416  EdgeMultiplayCallbacks.newRoomCreatedOnServer(newRoomCreatedOnServer.rooms);
417  break;
418 
419  default:
420  Debug.LogError("Unknown WebSocket message arrived: " + msg.type + ", message: " + message);
421  break;
422  }

425  void HandleUDPMessage(string message)
426  {
427  var msg = MessageWrapper.UnWrapMessage(message);
428  switch (msg.type)
429  {
430  case "GamePlayEvent":
431  GamePlayEvent mobiledgexEvent = Messaging<GamePlayEvent>.Deserialize(message);
432  if (mobiledgexEvent.eventName == "EdgeMultiplayObserver")
433  {
434  SyncObject(mobiledgexEvent);
435  }
436  else
437  {
438  // if used for other than that Syncing GameObjects Position & Rotation
439  // it wil trigger OnUDPMessagesReceived()
440  EdgeMultiplayCallbacks.udpEventReceived(mobiledgexEvent);
441  }
442  break;
443  default:
444  Debug.LogError("Unknown UDP message arrived: " + msg.type + ", message: " + message);
445  break;
446  }

449  void SyncObject(GamePlayEvent receivedEvent)
450  {
451  if (receivedEvent.senderId == gameSession.playerId)
452  {
453  return;
454  }
455  EdgeMultiplayObserver observedObject;
456  if (receivedEvent.booleanData[0]) // player object
457  {
458  observedObject = observers.Find(observer => observer.GetComponent<NetworkedPlayer>()
459  && observer.GetComponent<NetworkedPlayer>().playerId == receivedEvent.stringData[0]);
460  }
461  else
462  {
463  observedObject = observers.Find(observer => observer.eventId == receivedEvent.stringData[0]);
464  }
465  switch (receivedEvent.integerData[0])
466  {
467  case (int)EdgeMultiplayObserver.SyncOptions.SyncPosition:
468  observedObject.positionFromServer = new Vector3(receivedEvent.floatData[0], receivedEvent.floatData[1], receivedEvent.floatData[2]);
469  break;
470  case (int)EdgeMultiplayObserver.SyncOptions.SyncRotation:
471  observedObject.rotationFromServer = new Vector3(receivedEvent.floatData[0], receivedEvent.floatData[1], receivedEvent.floatData[2]);
472  break;
473  case (int)EdgeMultiplayObserver.SyncOptions.SyncPositionAndRotation:
474  observedObject.positionFromServer = new Vector3(receivedEvent.floatData[0], receivedEvent.floatData[1], receivedEvent.floatData[2]);
475  observedObject.rotationFromServer = new Vector3(receivedEvent.floatData[3], receivedEvent.floatData[4], receivedEvent.floatData[5]);
476  break;
477  }
